\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}
\tag{indeterminate form}

Let
\[
f(x) = \begin{cases}
  |x| &\text{if $x<1$,} \\
  \frac{x^2-a^2}{x-a} &\text{if $x>1$.}
\end{cases}
\]
If $\lim_{x \to 1} f(x)$ exists, what must be the value of $a$?
\begin{prompt}
  \[
a = \answer{0}
\]
\end{prompt}
\begin{hint}
  Equate the two one sided limits of $f$ at $x=1$ to obtain an equation involving $a$.
\end{hint}

\end{exercise}
