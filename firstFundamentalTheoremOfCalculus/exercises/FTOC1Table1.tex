\begin{exercise}

Let $g(x) = \int_2^{x^2} f(t) \d t$, and $f$ have the following table
of values.  What is $g'(3)$?

\[
\begin{array}{c|c}
 x & f(x) \\ \hline
1 & 3 \\
2 & 7 \\
3 & -5\\
4 & -8 \\
5 & 2 \\
6 & 6 \\
7 & 1 \\
8 & 3 \\
9 & 6 \\
10 & 8
\end{array}
\]
\begin{hint}
By the FFTC, $g'(x)=f(x^2)\cdot2x$.
\end{hint}
\begin{hint}
Therefore, $g'(3)=f(3^2)\cdot2(3)=6f(9)$.
\end{hint}
\begin{prompt}
  \[
  g'(3) = \answer{36}
  \]
\end{prompt}
\end{exercise}
