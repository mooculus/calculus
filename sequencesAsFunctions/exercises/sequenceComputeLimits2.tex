\begin{exercise}


Consider$\{a_n \}_{n=1}$ where $a_n = \sin(\left(\frac{1}{n}\right)^{n}$.  Then:
\[
\lim_{n \to \infty} a_n = \answer{0}
\]

\begin{hint}
The form of this limit is $(\answer{0})^{\answer{\infty}}$.  Is this indeterminate?
\end{hint}
\end{exercise}
