\begin{exercise}

\outcome{Compute cross products.}

\tag{cross products}
\tag{vectors}

Let $\vec{u}=\vector{4,-5,-5}$ and $\vec{v}=\vector{3,3,4}$. Compute the cross product $\vec{u} \cross \vec{v}$.

\begin{prompt}
\[
\vec{u}\cross\vec{v}=\vector{\answer{-5},\answer{-31},\answer{27}}
\]
\end{prompt}
