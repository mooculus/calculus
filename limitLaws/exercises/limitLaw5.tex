\begin{exercise}
\outcome{Calculate limits using the limit laws.}
\tag{limit}

	If you know that $\lim_{x \to 2} f(x) = 3$ and $\lim_{x \to 2} g(x) = 2$, then evaluate the following limit:
	
	\[
		\lim_{x \to 2} \left( 3 f(x)^2 + g(x) \right) = \answer{29}
	\]
	
\end{exercise}
