\begin{exercise}
  Let $C$ be the curve drawn from $A$ to $B$ below:
  \begin{image}
    \begin{tikzpicture}
      \begin{axis}%
        [
	  xmin=-7,xmax=7,
          ymin=-5,ymax=5,
          xlabel=$x$,ylabel=$y$,
          axis lines=center,
          every axis y label/.style={at=(current axis.above origin),anchor=south},
          every axis x label/.style={at=(current axis.right of origin),anchor=west},
          clip=false,
	  grid =major,
          xtick={-7,-6,...,7},
          ytick={-5,-4,...,5},
	]
        \addplot[penColor,ultra thick,smooth] coordinates{
          (6,3) (-4,4) (-5,-2) (2,-4)};
        \addplot[color=penColor,fill=penColor,only marks,mark=*]
        coordinates{(2,-4)};  %% closed hole
        \addplot[color=penColor,fill=penColor,only marks,mark=*]
        coordinates{(6,3)};  %% closed hole
        \node[right] at (axis cs:2,-4) {$A$};
        \node[right] at (axis cs:6,3) {$B$};
      \end{axis}
    \end{tikzpicture}
  \end{image}
  Compute:
  \[
  \int_C M\left(e^y\cos(x)\right)\d x + \left(e^y\sin(x)\right)\d y = \answer{e^3 \sin (6)-\frac{\sin (2)}{e^4}}
  \]
\end{exercise}
