\begin{exercise}
Let $Y(n) = -5 \sin (n) \left(\sqrt{n}+3 \sin (n)\right)$. Compute:
\[
Y'(n)
\begin{prompt}= -5 \sin(n) \left(\answer{(1/2)n^(-1/2) + 3\cos(n)}\right) + \left( \sqrt(n)+ 3\sin(n)\right) \left(\answer{-5 \cos(n)}\right) \end{prompt}
\]
\end{exercise}
