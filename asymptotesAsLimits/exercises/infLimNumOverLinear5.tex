\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\tag{limit}
Consider 
\[
f(\theta) = \frac{5}{\theta +5}.
\]
Compute
\begin{enumerate}
\item $\lim_{\theta\to -5^- } f(\theta) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{\theta\to -5^+ } f(\theta) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{\theta\to -5 } f(\theta) \begin{prompt} = \answer{DNE}\end{prompt}$
\end{enumerate}
\end{exercise}
