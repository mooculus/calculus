\begin{exercise}
Select all of the following that are true.

\begin{selectAll}
\choice[correct]{If $|x-3| \leq7$, then $-4 \leq x \leq 10$.}
\choice{If $|x+2| < 5$, then $|x|<3$.}
\choice{If $|(x-1)|^2 < 9$, then $x-1 < 3$.}
\choice[correct]{Solving the inequality $|x-a|<5$ requires that we find all $x$-values that are at most $5$ units away from $x=a$.}
\choice{Solving the inequality $|x-a|<5$ requires that we find all $x$-values that are at most $a$ units away from $x=5$.}
\end{selectAll}
\end{exercise}
