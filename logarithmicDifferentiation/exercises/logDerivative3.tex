\begin{exercise}

Calculate the following derivatives.
\begin{enumerate}
\item $\ddx\ln(x+\sqrt{x^2-1}) =\frac{1}{\answer{\sqrt{x^2-1}}}$\\
\item$ \ddx(\sin x)^{\tan{x}} = (\sin{x})^{\tan{x}}\left(1+\answer{\sec^2{(x)}\ln{(\sin{x})}}\right)$\\
\end{enumerate}
\end{exercise}
