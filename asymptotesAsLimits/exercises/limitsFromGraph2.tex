\begin{exercise}

The Graph of a function $f$ is given below.  The domain of $f$ is $\left(-\infty,\infty\right)$.

\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-20.3,xmax=20.3,ymin=-0.3,ymax=20.3,
        clip=true,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={0,5,...,20},
	xtick={-20,-10,...,20},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \addplot[very thick,penColor,domain=-4:-0.02] plot{10*(x+4)^2/(-x*(x+8))};
      \addplot[very thick,penColor,domain=-8:-4] plot{(x+4)^2/4};
      \addplot[very thick,penColor,domain=-20.3:-8] plot{64/(x+4)^2};
      \draw[very thick,penColor] (axis cs:0,5) -- (axis cs:8,5);
      \draw[very thick,penColor] (axis cs:8,10) -- (axis cs:20.3,10);
      
      \addplot[penColor,only marks,mark=*] coordinates{(-8,10)};
      \addplot[penColor,only marks,mark=*] coordinates{(0,5)};
      \addplot[penColor,only marks,mark=*] coordinates{(8,5)};
      
      \addplot[penColor,only marks,fill=white,mark=*] coordinates{(-8,4)};
      \addplot[penColor,only marks,fill=white,mark=*] coordinates{(8,10)};

      \node[penColor] at (axis cs:8, 17) [penColor] {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}

Find the following values.  Possible answers are any number, $\infty$, $-\infty$, and $DNE$.

\begin{align*}
\lim_{x\to8}f(x) &= \answer{DNE}\\
f(-8) &= \answer{10}\\
\lim_{x\to8^-}f(x) &= \answer{5}\\
\lim_{x\to0^-}f(x) &= \answer{\infty}
\end{align*}

\begin{exercise}

The intervals of continuity for $f$ from left to right are $\left(\answer{-\infty},\answer{-8}\right)$, $\left(\answer{-8},\answer{0}\right)$,$\left[\answer{0},\answer{8}\right]$, and $\left(\answer{8},\answer{\infty}\right)$.

\begin{exercise}

The only vertical asymptote of $f$ is $x=\answer{0}$.

\begin{exercise}

The horizontal asymptotes of $f$ from bottom to top are $y=\answer{0}$ and $y=\answer{10}$.

\end{exercise}
\end{exercise}
\end{exercise}
\end{exercise}
