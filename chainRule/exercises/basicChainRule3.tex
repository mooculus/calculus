\begin{document}
\begin{exercise}
Let $s(n) = \sec \left(\frac{2 \sqrt[3]{2}}{\sqrt[3]{n}}\right)$. Compute:

\[
\dd{n}s(n)
\begin{prompt}=\answer{-\frac{2 \sqrt[3]{2} \tan \left(\frac{2 \sqrt[3]{2}}{\sqrt[3]{n}}\right) \sec \left(\frac{2 \sqrt[3]{2}}{\sqrt[3]{n}}\right)}{3 n \sqrt[3]{n}}}\end{prompt}
\]
\end{exercise}
