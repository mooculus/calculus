\begin{exercise}
Suppose that $\sum_{k=1}^{\infty} a_k$ converges.  Which of the following \emph{must} be true?  

\begin{selectAll}
\choice[correct]{$\sum_{k=50}^{\infty} a_k$ must converge.}
\choice{$\lim_{n \to \infty} \sqrt[n]{a_n}$ must be strictly less than $1$.}
\choice{$\lim_{n \to \infty} \frac{a_{n+1}}{a_n}$ must be strictly less than $1$.}
\choice[correct]{$\{a_n\}$ must be bounded.}
\choice{$\{a_n\}$ is monotonic.}
\choice[correct]{$\lim_{n \to \infty} a_n = 0$}
\end{selectAll}

\begin{hint}
Consider $\sum_{k=1}^{\infty} \frac{1}{k^2}$.  
\begin{mulitpleChoice}
\choice{This is a geometric series.  It converges.}
\choice{This is a geometric series.  It diverges.}
\choice[correct]{This is a $p$-series.  It converges.}
\choice{This is a $p$-series.  It diverges.}
\end{multipleChoice}
What do Ratio Test and Root Test tell us if we apply them to this series?  
\end{hint}

\end{exercise}
