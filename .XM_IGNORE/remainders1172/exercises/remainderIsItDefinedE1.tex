\outcome{Understand the relationship between the sequence of remainders and the convergence of the series.}

\begin{document}

\begin{exercise}

Select all of following series for which it is possible to define remainders.

\begin{selectAll}
\choice[correct]{$\sum_{k=1}^{\infty} 2^{1-k}$}
\choice{$\sum_{k=1}^{\infty} \frac{2^{2k}}{3^k}$}
\choice{$\sum_{k=1}^{\infty} \bigg(\sin(k)-\sin(k+1) \bigg)$}
\choice[correct]{$\sum_{k=1}^{\infty}  \left(\frac{1}{k^2} - \frac{1}{(k+1)^2}\right)$}

\end{selectAll}

\begin{hint}
These are all geometric or telescoping series.  Remember that remainder terms are only defined if the series in question converges.
\end{hint}
\end{exercise}
