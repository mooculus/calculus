\begin{document}
\begin{exercise}

Consider
\[
xy- y^2-4=0.
\]
First compute
\[
\frac{dy}{dx} \begin{prompt}=\answer{\frac{3-3x-2y}{2x+y-1}}.
\end{prompt}
\]
Then compute

\[
\frac{d^2y}{dx^2}  \begin{prompt}=\answer{ \frac{1}{(2x+y-1)^2}}.
\end{prompt}
\]

\end{exercise}
