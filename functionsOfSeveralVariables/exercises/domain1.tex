\begin{exercise}
  Consider $F(x,y) = e^{x^2+y^2}$.
  \begin{itemize}
  \item The domain is:
    \[
    \answer{-\infty} < x < \answer{\infty}\quad\text{and}\quad \answer{-\infty} < y < \answer{\infty}
    \]
  \item The range is:
    \[
     \answer{1} < z < \answer{\infty}
    \]    
  \end{itemize}
\end{exercise}
