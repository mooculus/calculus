\begin{exercise}

Consider the region bounded by $y= -x^2+4x$ and
$y=x^2-6x+5$ over the interval $0 \le x \le 1$.


\begin{image}
\begin{tikzpicture}
	\begin{axis}[
            domain=-1:2, ymax=6,xmax=1.5,ymin=-0.3, xmin=-.5,

            xticklabels={$a$,$1$}, 
            every axis y label/.style={at=(current axis.above origin),anchor=south},
            every axis x label/.style={at=(current axis.right of origin),anchor=west},

          \addplot [draw=none,fill=fillp,domain=0.56:1] {-x^2+4*x} \closedcycle;
          \addplot [draw=none,fill=background,domain=0.56:1] {x^2-6*x+5} \closedcycle;

        \end{axis}

Note that the top curve changes since the two curves cross!  

How many integrals with respect to $x$ are needed to set up the area of the region? $\answer{2}$

\begin{exercise}

Find the $x$-value, $x=a$, of the intersection point of the curves.  $a =\answer{\frac{5-\sqrt{15}}{2}}$

\begin{hint}
Note that only one Of the two solutions is within the region of interest.
\end{hint}

\end{exercise}

\begin{exercise}

Set up a sum of integrals with respect to $x$ that express the area of the region: 
\[
\textrm{Area} = \int_0^a \answer{(x^2-6x+5)-(-x^2+4x) } \d x + \int_a^1 \answer{ (-x^2+4x)-(x^2-6x+5) } \d x
\]
\end{exercise}
