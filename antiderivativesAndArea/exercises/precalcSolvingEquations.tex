\begin{exercise}
Solve the following equation for $x$.
\[
\sqrt{4-x}-\frac{x}{2\sqrt{4-x}}=0
\]
\begin{hint}
Multiply the equation by the common denominator, $2\sqrt{4-x}$.
\end{hint}
ANSWER: 
\[
x=\answer{\frac{8}{3}}
\]
\end{exercise}
