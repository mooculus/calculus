\begin{exercise}

\outcome{Answer questions about lines and curves in space.}

\tag{curves}


Let $\vecl_{1}(t)=\vector{1,1,1}+t\vector{3,1,3}$ and $\vecl_{2}(t)=\vector{7,3,7}+t\vector{6,2,6}$.

Then $\vecl_{1}$ and $\vecl_{2}$ are: 

\begin{multipleChoice}
\choice{parallel}
\choice{intersecting}
\choice{non-intersecting}
\choice[correct]{the same}
\end{multipleChoice}


\end{exercise}
