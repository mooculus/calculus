Consider the slope field below:

\begin{image}
{\def\length{sqrt(1+((x-y)/(x+y))^2)}
\begin{tikzpicture}
  \begin{axis}[
      xmin=-3, xmax=3,ymin=-3,ymax=3,domain=-3:3,view={0}{90},
      axis lines =center, xlabel=$x$, ylabel=$y$,
      every axis y label/.style={at=(current axis.above origin),anchor=south},
      every axis x label/.style={at=(current axis.right of origin),anchor=west},
      axis on top,
    ] 
    \addplot3 [penColor, quiver={u={1/\length}, v={(x-y)/(x+y)/(\length)},scale arrows=.2},samples=20] {0};
]  \end{axis}
\end{tikzpicture}}
\end{image}


\begin{multipleChoice} 
\choice{$\frac{dy}{dx} = \frac{x-y}{x+y}$}
\choice[correct]{$\frac{dy}{dx} = \frac{x+y}{x-y}$}
\choice{$\frac{dy}{dx} = \frac{x-y}{x^2+y^2}$}
\choice{$\frac{dy}{dx} = \frac{y-x}{x^2+y^2}$}
\end{multipleChoice}

\end{exercise}
