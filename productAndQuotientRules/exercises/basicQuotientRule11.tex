\begin{exercise}

\[
f'(x)
\begin{prompt}=\answer{\frac{2 e^x-\sqrt[5]{x}}{15 x \sqrt[5]{x}}-\frac{2 e^x-\frac{1}{5 \sqrt[5]{x}^4}}{3 \sqrt[5]{x}}}\end{prompt}
\]
\end{exercise}
