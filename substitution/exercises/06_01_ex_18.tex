\begin{exercise}

Find the indefinite integral of $\frac{e^{\sqrt{x}}}{\sqrt{x}}$ with respect to $x$.

\[
\int \frac{e^{\sqrt{x}}}{\sqrt{x}}\d x=\answer{2e^{\sqrt{x}}}+C
\]

\begin{hint}
	Try the substitution $w=\sqrt{x}$.
\end{hint}
