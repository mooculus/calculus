\begin{exercise}
  Consider the vectors below:
  \begin{image}
    \begin{tikzpicture}
      \begin{axis}[
          xmin=-5,xmax=5,ymin=-5,ymax=5,
          clip=false,
          unit vector ratio*=1 1 1,
          %axis lines=none,
          grid = major,
          ytick={-5,-4,...,5},
	  xtick={-5,-4,...,5},
          ticks = none,
        ]
          \addplot[very thick,penColor,->] plot coordinates {(0,0) (3.54,.66)};
          \addplot[very thick,penColor2,->] plot coordinates {(0,0) (1.62,3.14)};
          \node[below] at (axis cs:1.75, .33) [penColor] {$\vec{v}$};
          \node[left] at (axis cs:.8, 1.6) [penColor2] {$\vec{w}$};
        \end{axis}
    \end{tikzpicture}
  \end{image}
  Assuming that the components of $\proj_{\vec{v}}(\vec{w})$ are whole
  numbers,
  \[
  \proj_{\vec{v}}(\vec{w}) = \vector{\answer{1},\answer{2}}
  \]
  You should be able to answer this question \textbf{without} any
  computation.  
\end{exercise}
