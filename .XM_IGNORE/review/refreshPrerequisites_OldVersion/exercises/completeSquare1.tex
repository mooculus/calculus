\begin{exercise}  
  Find the missing quantities that complete the squares:
  \[
  x^2+20x = x^2 + 20x +\answer{100} - \answer{100} = \left(x+\answer{10}\right)^2 -\answer{100}
   \]

  \[
  x^2-6x = x^2 - 6x +\answer{9} - \answer{9} = \left(x+\answer{3}\right)^2 -\answer{9}
   \]
   
\end{exercise}

  \]
 \end{exercise}
