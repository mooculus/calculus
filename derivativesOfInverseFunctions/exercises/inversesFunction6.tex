\begin{exercise}

If $f(2) = 4$ and $f(2.1) = 4.3$, estimate $(f^{-1})'(4)$.

\begin{hint}
Recall:  $(f^{-1})'(4)=\frac{1}{f'(2)}$
\end{hint}

\begin{hint}
Recall:  $f'(2)\approx\frac{f(2+h)-f(2)}{h}$
\end{hint}

\begin{prompt}
\[
(f^{-1})'(4) \approx \answer{\frac{1}{3}}
\]
\end{prompt}
\end{exercise}
