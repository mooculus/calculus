\begin{exercise}
Let $f(x) = \frac{ 3 x^2+2x}{4x-3 }$. Compute:
\[
f'(x))
\begin{prompt}=\answer{\frac{(6x+2)(4x-3)-(3 x^2+2x)(4)}{(4x-3)^2}}\end{prompt}
\]
(No need to simplify)
\end{exercise}
