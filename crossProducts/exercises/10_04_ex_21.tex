\begin{exercise}

\outcome{Compute cross products.}

\tag{cross products}
\tag{vectors}

Let $\vec{u}=\vector{-2,1,5}$ and $\vec{v}=\vector{-1,3,1}$. Compute $|\vec{u}\cross\vec{v}|$.

\begin{prompt}
\[
|\vec{u}\cross\vec{v}|=\answer{\sqrt{230}}
\]
\end{prompt}
