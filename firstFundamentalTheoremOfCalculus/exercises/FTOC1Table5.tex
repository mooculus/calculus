\begin{exercise}

Let $g(x) =  \left[ \int_0^{x} f(t) \d t \right]^2$, and $h(x) = \int_0^{x} f(t) dt$.  $f$ and $h$ have the following table of values.  What is $g'(3)$?

\[
\begin{array}{c|c|c}
 x & f(x)  & h(x)\\ \hline
1 & 3 & -1\\
2 & 7 & 0\\
3 & -5 & 3\\
4 & -8& 2 \\
5 & 2 & 9\\
6 & 6 & 5\\
7 & 1 & 4\\
8 & 3& 2 \\
9 & 6 & -4\\
10 & 8& -9\\
\end{array}
\]

\begin{prompt}
	\[
		g'(3) = \answer{-30}
	\]
\end{prompt}

\end{exercise}
