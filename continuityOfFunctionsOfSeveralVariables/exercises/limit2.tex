\outcome{Evaluate limits of functions of several variables.}

\begin{exercise}
  Compute the following limit. If it does not exist, write $\verb|DNE|$.
  \[
  \lim_{\vector{x,y}\to \vector{0,0}}\frac{x^2+4xy+4y^2}{2x+4y} = \answer{0}
  \]
\end{exercise}
