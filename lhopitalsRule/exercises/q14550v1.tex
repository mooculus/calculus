\begin{exercise}

Evaluate the limit: \[\lim_{x \to -\infty} x^5e^{2x}.\]


\begin{hint}
This is a L'Hopital's rule problem, of the indeterminate form
$-\infty \cdot 0.$\\
Try to rewrite the function so to be able to apply L'Hospital rule. 
\end{hint}


\begin{multipleChoice}
\choice{$\frac{15}{4}$}
\choice{$\infty$}
\choice[correct]{$0$}
\choice{$-\infty$}
\choice{DNE}
\end{multipleChoice}

\end{exercise}
