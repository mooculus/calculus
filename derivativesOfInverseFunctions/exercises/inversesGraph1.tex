\begin{exercise}

The entire graph of the function $f$ is given below.

\begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-5.3,xmax=5.3,ymin=-5.3,ymax=5.3,
        clip=true,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={-5,-4,...,5},
    xtick={-5,-4,...,5},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
      \draw[very thick,penColor] (axis cs:-4,4) -- (axis cs:-1,1);
      \draw[very thick,penColor] (axis cs:1,-2) -- (axis cs:4,0);
      \addplot[very thick,penColor,only marks,mark=*] coordinates{(-4,4) (-1,1) (1,-2) (4,0)};
      
      \node at (axis cs:3.5,3.5) [penColor] {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
\end{image}

Calculate the following values or write "DNE" (does not exist).
\begin{align*}
(f^{-1})'(3) &= \answer{-1}\\
(f^{-1})'(4) &= \answer{DNE}\\
(f^{-1})'(-0.5) &= \answer{\frac{3}{2}}\\
(f^{-1})'(-4) &= \answer{DNE}
\end{align*}
