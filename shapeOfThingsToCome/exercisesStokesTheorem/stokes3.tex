\begin{exercise}
  Let $P$ be a parallelogram (drawn in a counterclockwise fashion)
  with vertices:
  \[
  (0,0,0)\quad (3,2,1)\quad (2,3,1) \quad (5,5,2)
  \]
  Compute:
  \[
  \oint_P (x^2-z) \d x + (y^3+x)\d y + y \d z
  \begin{prompt}
    = \answer{5}
  \end{prompt}
  \]
\end{exercise}
