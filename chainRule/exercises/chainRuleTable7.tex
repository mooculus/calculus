\begin{exercise}

Let $h(x) = g(\left[f(x)\right]^2)$, where $f$ and $g$ are functions which have the properties shown in the table below.  Evaluate $h'(9)$.

\[
\begin{array}{c|c|c|c|c}
 x & f(x) & f'(x) & g(x) & g'(x)\\ \hline
1 & -4 & -1 & -7 & -9\\
2 & 6 & -8 & 6 & -6\\
3 & 8 & -6 & 8 & -8\\
4 & -1 & 5 & -5 & -9\\
5 & 8 & -1 & 7 & 1\\
6 & 3 & -3 & -5 & -6\\
7 & -5 & 0 & -9 & 6\\
8 & 6 & 6 & 3 & 4\\
9 & 2 & -5 & 6 & -3\\
10 & -3 & 2 & -7 & 3\\
\end{array}
\]


\begin{prompt}
	$$h'(9) = \answer{18}$$
\end{prompt}


\end{exercise}
