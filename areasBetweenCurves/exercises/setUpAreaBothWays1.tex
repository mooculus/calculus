\begin{exercise}
	The region $R$ bounded by $y=3-x^2$ and $y=x+1$ is shown below: 

   \begin{image}
            \begin{tikzpicture}
            	\begin{axis}[
            		domain=-2.6:1.6, ymax=3.4,xmax=1.6, ymin=-2.4, xmin=-2.6,
            		axis lines =center, xlabel=$x$, ylabel=$y$,
            		every axis y label/.style={at=(current axis.above origin),anchor=south},
            		every axis x label/.style={at=(current axis.right of origin),anchor=west},
            		axis on top,
            		]
                      
            	\addplot [draw=penColor,very thick,smooth] {3-x^2};
            	\addplot [draw=penColor2,very thick,smooth] {x+1};
	                            
            	\addplot [name path=A,domain=-2:1,draw=none] {3-x^2)};   
            	\addplot [name path=B,domain=-2:1,draw=none] {x+1};
            	\addplot [fillp] fill between[of=A and B];
                      
                      
            	\node at (axis cs:-1.5,2.5) [penColor] {$y=3-x^2$};
            	\node at (axis cs:.7,1) [penColor2] {$y=x+1$};
	    
	      \end{axis}
            \end{tikzpicture}
            \end{image}

Set up, but DO NOT EVALUATE, an integral or sum of integrals with respect to $x$ that would compute the area of $R$. 

\begin{multipleChoice}
\choice[correct]{One integral is needed to compute the area.  

In fact, $A=\int_{\answer{-2}}^{\answer{1}} \answer{(3-x^2)-(x+1)} \d x$}.
\choice{Two integrals are needed to compute the area.  

In fact, $A=\int_{-2}^{\answer{\pi/3}} \answer{\sin(2x)-\sin(x)} \d x+\int_{\answer{\pi/3}}^{\answer{\pi}} \answer{\sin(x)-\sin(2x)} \d x$}. 
\end{multipleChoice}

Set up, but DO NOT EVALUATE, an integral or sum of integrals with respect to $y$ that would compute the area of $R$. 

\begin{multipleChoice}
\choice{One integral is needed to compute the area.  

In fact, $A=\int_{\answer{-2}}^{\answer{1}} \answer{(3-x^2)-(x+1)} \d x$}.
\choice[correct]{Two integrals are needed to compute the area.  

In fact, $A = \int_{-1}^{\answer{2}} \answer{(y-1)-(-\sqrt{3-y})} \d y  +\int_{\answer{2}}^{\answer{3}} \answer{\sqrt{3-y}-(-\sqrt{3-y})} \d y $}.
\end{multipleChoice}
	
\end{exercise}
