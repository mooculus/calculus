\begin{exercise}

\outcome{Answer questions about lines and curves in space.}

\tag{curves}


Let $\vecl_{1}(t)$ be the line whose components are $\left\{ \begin{aligned}
x(t) & =0.2+0.6t\\
y(t) & =1.33-0.45t\\
z(t) & =-4.2+1.05t
\end{aligned}
\right.$  and $\vecl_{2}(t)$ be the line whose components are $\left\{ \begin{aligned}
x(t) & =0.86+9.2t\\
y(t) & =0.835-6.9t\\
z(t) & =-3.045+16.1t
\end{aligned}
\right.$

Then $\vecl_{1}$ and $\vecl_{2}$ are: 

\begin{multipleChoice}
\choice{parallel}
\choice{intersecting}
\choice{non-intersecting}
\choice[correct]{the same}
\end{multipleChoice}


\end{exercise}
