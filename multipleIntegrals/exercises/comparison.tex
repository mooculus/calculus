\outcome{Compare the relative sizes of two multiple integrals.}

\begin{document}

\begin{exercise}
  Let $S$ be $[0,1] \times [0,1] \times [0,1]$.
  
  Let $F : S \to \R$ be the function defined by the rule
  \[
    F(x,y,z) = x y^2 z^3.
  \]
  and let $G : S \to \R$ be the function defined by the rule
  \[
    G(x,y,z) = x y z.
  \]
  Which integral is larger?
  \begin{multipleChoice}
    \choice{$\iiint_S F \d V$}
    \choice[correct]{$\iiint_S G \d V$}    
  \end{multipleChoice}
  
  \begin{hint}
    Note that, when $(x,y,z) \in S$, we have $F(x,y,z) \leq G(x,y,z)$.
  \end{hint}
  
\end{exercise}
