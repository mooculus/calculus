\begin{exercise}
Recall that the series:

\[
\sum_{k=k_0}^{\infty} a_k
\]

is called \emph{geometric} is the sequence $\{a_n\}_{n=n_0}$ is a geometric sequence; i.e. that  $a_k$ can be brought into the form:

\[
a_k = ar^k
\]
for some constants $a$ and $r$.

\begin{exercise}
A geometric series converges if:
\begin{multipleChoice}
\choice{$r<1$}
\choice[correct]{$|r|<1$}
\choice{$|r|\leq1$}
\choice{$|r|\geq1$}
\choice{$|r|>1$}
\end{multipleChoice}

\end{exercise}
How does the value of the constant $a$ affect the series?
\begin{multipleChoice}
\choice[correct]{The value of $a$ does not affect whether the series converges, but if the series converges, it will affect the value to which it converges.}
\choice{The value of $a$ will affect whether the series converges or diverges.}
\end{multipleChoice}

How does the value of the lower index $k_0$ affect the series?
\begin{multipleChoice}
\choice[correct]{The value of $k_0$ does not affect whether the series converges, but if the series converges, it will affect the value to which it converges.}
\choice{The value of $k_0$ will affect whether the series converges or diverges.}
\end{multipleChoice}

\begin{exercise}
If $|r|<1$, then the geometric series $\sum_{k=0}^{\infty} ar^k$ converges to $\answer{\frac{a}{1-r}}$.

\begin{exercise}
If $|r|<1$, then the geometric series $\sum_{k=k_0}^{\infty} ar^k$ converges to $\answer{\frac{ar^{k_0}}{1-r}}$.

Does this agree with the above result if $k_0=0$?

\begin{multipleChoice}
\choice[correct]{Yes}
\choice{No}
\end{multipleChoice}
\end{exercise}
\end{exercise}


\end{exercise}
