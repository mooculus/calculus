\begin{exercise}

\tag{derivative}

\outcome{Apply the generalized power rule.}
\outcome{Take derivatives of functions raised to functions.}
\outcome{Use order of operations in situations requiring multiple derivative rules.}


Find the derivative of $m(w)=\frac{3^w}{2^w}$ with respect to $w$.
\begin{prompt}
  \[
  m'(w) = \answer{\ln(3/2)(3/2)^w}
  \]
\end{prompt}

\end{exercise}
