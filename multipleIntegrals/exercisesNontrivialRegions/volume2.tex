\outcome{Use integrals to define the volume of a region.}

\begin{document}

\begin{exercise}
  Let $B$ be the region $\{ (x,y,z) \in \R^3 : x^2 + y^2 + z^2 \leq 1 \}$

  Evaluate $\iiint_B 1 \d V$.
  \begin{prompt}
    \[
      \iiint_B 1 \d V = \answer{4 \pi / 3}.
    \]    
  \end{prompt}

  \begin{exercise}
    Let $C$ be the region $\{ (x,y,z) \in \R^3 : x^2 + 2y^2 + 3z^2 \leq 1 \}$.

    How does $\iiint_C 1 \d V$ compare to $4 \pi / 3$?
    \begin{multipleChoice}
      \choice[correct]{$\iiint_C 1 \d V < 4 \pi / 3$}
      \choice{$\iiint_C 1 \d V = 4 \pi / 3$}
      \choice{$\iiint_C 1 \d V > 4 \pi / 3$}      
    \end{multipleChoice}
  \end{exercise}
  
\end{exercise}
