\begin{exercise}
Let $B(x) = \frac{ -\frac{2}{x^4}-3 \sqrt{x}}{-2 e^x}$. Compute:
\[
\dd{x}B(x)
\begin{prompt}=\answer{\frac{1}{2} e^{-x} \left(-\frac{2}{x^4}-3 \sqrt{x}\right)-\frac{1}{2} e^{-x} \left(\frac{8}{x^5}-\frac{3}{2 \sqrt{x}}\right)}\end{prompt}
\]
\end{exercise}
