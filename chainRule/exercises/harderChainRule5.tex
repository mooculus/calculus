\begin{exercise}
Let $g(w) = 2 \sqrt{e^{\frac{3}{\sqrt[3]{2} \sqrt[3]{w}}}}$. Compute:

\[
\dd{w}g(w)
\begin{prompt}=\answer{-\frac{\sqrt{e^{\frac{3}{\sqrt[3]{2} \sqrt[3]{w}}}}}{\sqrt[3]{2} w \sqrt[3]{w}}}\end{prompt}
\]
\end{exercise}
