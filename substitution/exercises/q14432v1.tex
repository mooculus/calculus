\begin{exercise}

Find the antiderivative: \[\int 2e^{4x}\; dx.\]





\begin{multipleChoice}
\choice{$2e^{4x}+C$}
\choice{$8e^{4x}+C$}
\choice{$8e^{4x}$}
\choice{$2e^{4x}$}
\choice[correct]{$\frac{1}{2}e^{4x}+C$}
\end{multipleChoice}

\end{exercise}
