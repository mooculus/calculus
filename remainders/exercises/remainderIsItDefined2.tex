\outcome{Understand the relationship between the sequence of remainders and the convergence of the series.}

\begin{document}

\begin{exercise}

Select all of following series for which it is possible to define a sequence of remainders.

\begin{selectAll}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{(-1)^k}{\sqrt{k}}$}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{4k}{(k^3+9)^2}$}
\choice{$\sum_{k=1}^{\infty} (-1)^k \cdot \frac{2k}{k+4}$}
\choice[correct]{$\sum_{k=1}^{\infty}  \left(\frac{3}{5}\right)^k$}
\end{selectAll}


\end{exercise}
