\begin{exercise}
Since the described slices are perpendicular to the $x$-axis, the slices are:
\begin{multipleChoice}
\choice[correct]{vertical}
\choice{horizontal}
\end{multipleChoice}

Hence, we integrate:
\begin{multipleChoice}
\choice[correct]{with respect to $x$.}
\choice{with respect to $y$.}
\end{multipleChoice}

Since the cross section is a square, we let one of its sides lie on the base of the $xy$-plane, while the other extend vertically above the $xy$-plane.  We want to use the formula:

\[ 
V= \int_{x=a}^{x=b} A \d x
\]

To do this, we must express the limits of integration and the cross-sectional area in terms of $x$.

\begin{exercise}
The limits of integration are $a= \answer{-2}$ and $b= \answer{2}$.           

\begin{hint}
To find these, set $8-x^2=x^2$ and solve for $x$.
\end{hint}
 
\begin{exercise}
The cross-sectional area in questions is that of a square, so $A = s^2$.  We must express $s$ in terms of the variable of integration $x$.  Since $s$ is a vertical distance, we have $s=y_{top}-y_{bot}$.

Choose the correct curve for $y_{top}$:
\begin{multipleChoice}
\choice[correct]{$y_{top}=8-x^2$}
\choice{$y_{top}=x^2$}
\end{multipleChoice}

Choose the correct curve for $y_{bot}$:
\begin{multipleChoice}
\choice{$y_{bot}=8-x^2$}
\choice[correct]{$y_{bot}=x^2$}
\end{multipleChoice}

So, $s= (8-x^2)-x^2$.
\end{exercise}


\begin{exercise}
Now, using the formula $V= \int_{x=a}^{x=b} A \d x$, we find:

\[
 V= \int_{x=\answer{-2}}^{x=\answer{2}} \answer{(8-2x^2)^2} \d x
\]

and upon evaluating this integral, we find the volume is $\answer{ 2048/15 }$.
\end{exercise}

\end{exercise}
\end{exercise}
