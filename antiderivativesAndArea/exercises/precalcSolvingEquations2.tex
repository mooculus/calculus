\begin{exercise}
Solve the following equation for x.
\[
\ln{(x^2+x)}-\ln{x}=4
\]
\begin{hint}
Use the properties of logarithms.
\end{hint}

ANSWER: 
\[
x=\answer{e^4-1}
\]
\end{exercise}
