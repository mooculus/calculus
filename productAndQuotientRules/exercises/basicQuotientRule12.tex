\begin{exercise}
Let $c(z) = \frac{ 3 \sqrt{z }-3 \sin (z )}{-e^{z }}$. Compute:
\[
c'(z)
\begin{prompt}=\answer{e^{-z } \left(3 \sqrt{z }-3 \sin (z )\right)-e^{-z } \left(\frac{3}{2 \sqrt{z }}-3 \cos (z )\right)}\end{prompt}
\]
(No need to simplify)
\end{exercise}
