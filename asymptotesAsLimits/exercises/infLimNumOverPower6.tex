\begin{exercise}
\outcome{Define a vertical asymptote.}
\outcome{Calculate limits of the form number over zero.}
\tag{limit}
Consider 
\[
f(v) = \frac{-1}{(v-3)^3}.
\]
Compute
\begin{enumerate}
\item $\lim_{v\to 3^- } f(v) \begin{prompt} = \answer{\infty}\end{prompt}$
\item $\lim_{v\to 3^+ } f(v) \begin{prompt} = \answer{-\infty}\end{prompt}$
\item $\lim_{v\to 3 } f(v) \begin{prompt} = \answer{DNE}\end{prompt}$
\end{enumerate}
\end{exercise}
