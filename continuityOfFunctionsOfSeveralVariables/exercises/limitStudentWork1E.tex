\begin{document}
\begin{exercise}
  Consider $F(x,y) = \frac{6xy}{5x^2+4y^2}$. While computing
  \[
  \lim_{\point{x,y}\to \point{0,0}} F(x,y)
  \]
  a student provides the following response:
  \begin{image}
    \includegraphics{studentWork.png}
  \end{image}
  \begin{multipleChoice}
    \choice{The student's work is correct.}
    \choice[correct]{The student's work is incorrect.}
  \end{multipleChoice}
  \begin{exercise}
    Evaluate the limit along the pth $y=mx$.
    \[
    \lim_{x\to 0 }F(x,mx) = \answer{\frac{6 m}{5+4m^2}}
    \]
    \begin{exercise}
      So we conclude:
      \begin{multipleChoice}
        \choice{The limit exists and is equal to $\frac{6 m}{5+4m^2}$.}
        \choice[correct]{The limit does not exist.}
      \end{multipleChoice}
    \end{exercise}
  \end{exercise}
\end{exercise}
