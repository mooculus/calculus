\begin{exercise}
  Let $f$ be a function defined on $(-8,8)$. The graph of $f$ is given in the figure below.
 
  \begin{image}
  \begin{tikzpicture}
    \begin{axis}[
        xmin=-8,xmax=8,ymin=-8,ymax=8,
        clip=false,
        unit vector ratio*=1 1 1,
        axis lines=center,
        grid = major,
        ytick={-8,...,8},
	xtick={-8,...,8},
        xlabel=$x$, ylabel=$y$,
        every axis y label/.style={at=(current axis.above origin),anchor=south},
        every axis x label/.style={at=(current axis.right of origin),anchor=west},
      ]
    \addplot[very thick,penColor,domain=-2:0] {(-x^2)};
      \addplot[very thick,penColor,domain=0:2] {(x^2)};
  
         \addplot[very thick,penColor,domain=4:8] {(3/2)*x^2-18*x+48};
      \addplot[very thick,penColor,domain=2:4] {2*sqrt(abs(2*(x-4)))};
       \addplot[very thick,penColor,domain=-4:-2] {-2*sqrt(abs(2*(-x-4)))};
      \addplot[very thick,penColor,domain=-8:-4] {-2*sqrt(abs(2*(-x-4)))};
      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(-2,-4)};  %% open hole
      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(-8,-5.6)};  %% open hole

      \addplot[color=penColor,fill=white,only marks,mark=*] coordinates{(8,0)};  %% open hole
      \addplot[color=penColor,fill=penColor,only marks,mark=*] coordinates{(-2,0)};  %% closed hole  
      
      \node[penColor] at (axis cs:4, 6) [penColor] {$y=f(x)$};
      \end{axis}`
  \end{tikzpicture}
  \end{image}
  Select  all values of $x$ where the function $f$ has a  critical point.
  
\begin{selectAll}
\choice{$x=-8$}
\choice{$x=-7$}
\choice[correct]{$x=-4$}
\choice[correct]{$x=-2$}
\choice[correct]{$x=0$}
\choice{$x=1$}
\choice[correct]{$x=2$}
\choice[correct]{$x=4$}
\choice[correct]{$x=6$}
\choice{$x=8$}
\end{selectAll}
