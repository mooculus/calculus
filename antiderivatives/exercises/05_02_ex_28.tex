\begin{document}
\begin{exercise}

Find the indefinite integral of $t^{1/3}+\frac{1}{t^2}+2^t$ with respect to $t$.
\begin{prompt}
  \[
  \int t^{1/3}+\frac{1}{t^2}+2^t\d t
  =\answer{\frac{3}{4}t^{4/3}-\frac{1}{t}+\frac{1}{\ln(2)}2^{t}+C}
  \]
\end{prompt}
\end{exercise}
