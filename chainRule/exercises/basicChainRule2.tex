\begin{exercise}
Let $B(u) = -\frac{1}{\sqrt[5]{2} \sqrt[5]{\csc (2 u)}}$. Compute:

\[
\dd{u}B(u)
\begin{prompt}=\answer{-\frac{2^{4/5} \cot (2 u)}{5 \sqrt[5]{\csc (2 u)}}}\end{prompt}
\]
\end{exercise}
