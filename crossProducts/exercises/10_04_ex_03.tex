\begin{exercise}

\outcome{Define a cross product.}

\tag{cross products}
\tag{vectors}
\tag{true or false}


Is the following statement true or false?

\begin{quote}
A fundamental principle of the cross product is that $\vec{u}\cross\vec{v}$ is orthogonal to $\vec{u}$ and $\vec{v}$.
\end{quote}

\begin{multipleChoice}
\choice[correct]{True}
\choice{False}
\end{multipleChoice}

\end{exercise}
