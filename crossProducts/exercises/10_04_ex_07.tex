\begin{exercise}

\outcome{Compute cross products.}

\tag{cross products}
\tag{vectors}

Let $\vec{u}=\vector{3,2,-2}$ and $\vec{v}=\vector{0,1,5}$. Compute the cross product $\vec{u} \cross \vec{v}$.

\begin{prompt}
\[
\vec{u}\cross\vec{v}=\vector{\answer{12},\answer{-15},\answer{3}}
\]
\end{prompt}
