Let $F(x,y) = \frac{x-y^2}{x^2-y^4}$.

\begin{problem}
  State the domain of $F(x,y)$.
  \begin{prompt}
    The domain is all real numbers except those where
    $\answer{x^2-y^4} = 0$.
  \end{prompt}
\end{problem}


\begin{problem}
  Does $\lim_{\vector{x,y}\to \vector{1,-1}} F(x,y)$
  exist? If so, compute the limit; if not, justify your answer.
  \begin{prompt}
    \begin{multipleChoice}
      \choice[correct]{The limit exists.}
      \choice{The limit does not exist.}
    \end{multipleChoice}
    \begin{problem}
      \[
      \lim_{\vector{x,y}\to \vector{1,-1}} F(x,y) =\answer{1/2}
      \]
    \end{problem}
  \end{prompt}
\end{problem}


\begin{problem}
  Consider $G(x,y) = F(x,y)$ when $(x,y) \ne (1,-1)$. Can one set a
  value for $G(1,-1)$ that will ensure $G$ is continuous at
  $(1,-1)$. Why or why not?
  \begin{prompt}
  \begin{multipleChoice}
    \choice[correct]{yes}
    \choice{no}
  \end{multipleChoice}
  \begin{problem}
    Set $G(1,-1) = \answer{1/2}$.
    \begin{feedback}
      Now $\lim_{\vector{x,y}\to\vector{1,-1}} G(x,y) = G(1,-1)$,
      hence $G$ is continuous at $(1,-1)$.
    \end{feedback}
  \end{problem}
  \end{prompt}
\end{problem}

\begin{problem}
  Let $\vec{v}$ and $\vec{w}$ be nonzero vectors and let $\vec{f}$ be
  a differentiable vector-valued function. Select statments that must be true:
  \begin{selectAll}
    \choice{If $|\vec{f}(t)| = 2$ and $|\vec{f}'(t)| = 3$, then $|\vec{f}(t) \dotp \vec{f}'(t)| = 6$.}
    \choice[correct]{If $|\vec{f}(t)| = 2$ and $|\vec{f}'(t)| = 3$, then $|\vec{f}(t) \cross \vec{f}'(t)| = 6$.}
    \choice{$\vec{v} \cross \vec{w} = |\vec{v}| \cdot |\vec{w}| \cdot \sin \theta$.}
    \choice{$\left|\dd{t} \vec{f}(t) \right| = \dd{t} | \vec{f}(t)|$.}
    \choice{$|\vec{v} + \vec{w}| = |\vec{v}| + |\vec{w}|$.}
    \choice{None of the above.}
\end{selectAll}
\end{problem}
