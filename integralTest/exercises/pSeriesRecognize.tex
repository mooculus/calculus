\begin{exercise}
Select all of the following that are $p$-series:
\begin{selectAll}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{1}{k^2}$}
\choice{$\sum_{k=1}^{\infty} \frac{2^k}{k^3}$}
\choice{$\sum_{k=1}^{\infty} \frac{1}{p^k}$}
\choice{$\sum_{k=1}^{\infty} k^{3/2}$}
\choice[correct]{$\sum_{k=1}^{\infty} \frac{1}{\sqrt[4]{k}}$}
\choice{$\sum_{k=1}^{\infty} \left(\frac{2}{3}\right)^k$}
\end{selectAll}
\end{exercise}
